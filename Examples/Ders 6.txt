#include <stdio.h>
#include <stdbool.h>

void printDiamond(int n)
{
    bool flag = true;
    int x = 0;

    for(int k = 0;k < 2 * n - 1;++k){ //for(;;)
        for(int i = 0;i < n - x - 1;++i)
            printf(" ");
        for(int i = 0;i < 2 * x + 1;++i)
            printf("*");
        if(x == n - 1)
            flag = false;
        if(flag)
            ++x;
        else
            --x;
        //if(x < 0)
            //break;
        printf("\n");
    }
}

int main()
{
    int n;
    printf("Enter n?");
    scanf("%d",&n);
    printDiamond(n);
}
